-- schema
DROP DATABASE IF EXISTS toysdb;
CREATE DATABASE toysdb CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE toysdb;

CREATE TABLE Role (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(60) NOT NULL UNIQUE
) ENGINE=InnoDB;

CREATE TABLE User (
  id INT PRIMARY KEY AUTO_INCREMENT,
  role_id INT NOT NULL,
  fio VARCHAR(120) NOT NULL,
  login VARCHAR(120) NOT NULL UNIQUE,
  password VARCHAR(120) NOT NULL,
  FOREIGN KEY (role_id) REFERENCES Role(id)
) ENGINE=InnoDB;

CREATE TABLE Category (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(80) NOT NULL UNIQUE
) ENGINE=InnoDB;

CREATE TABLE Manufacturer (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(80) NOT NULL UNIQUE
) ENGINE=InnoDB;

CREATE TABLE Supplier (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(80) NOT NULL UNIQUE
) ENGINE=InnoDB;

CREATE TABLE Unit (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(20) NOT NULL UNIQUE
) ENGINE=InnoDB;

CREATE TABLE Product (
  article VARCHAR(20) PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  unit_id INT NOT NULL,
  price DECIMAL(10,2) NOT NULL CHECK (price >= 0),
  supplier_id INT NOT NULL,
  manufacturer_id INT NOT NULL,
  category_id INT NOT NULL,
  discount INT NOT NULL DEFAULT 0 CHECK (discount >= 0 AND discount <= 100),
  stock INT NOT NULL DEFAULT 0 CHECK (stock >= 0),
  description TEXT,
  photo VARCHAR(255),
  FOREIGN KEY (unit_id) REFERENCES Unit(id),
  FOREIGN KEY (supplier_id) REFERENCES Supplier(id),
  FOREIGN KEY (manufacturer_id) REFERENCES Manufacturer(id),
  FOREIGN KEY (category_id) REFERENCES Category(id)
) ENGINE=InnoDB;

CREATE TABLE PickupPoint (
  id INT PRIMARY KEY AUTO_INCREMENT,
  address VARCHAR(255) NOT NULL UNIQUE
) ENGINE=InnoDB;

CREATE TABLE OrderStatus (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(40) NOT NULL UNIQUE
) ENGINE=InnoDB;

CREATE TABLE `Order` (
  id INT PRIMARY KEY AUTO_INCREMENT,
  article VARCHAR(255) NOT NULL,
  status_id INT NOT NULL,
  pickup_id INT NOT NULL,
  client_id INT NULL,
  order_date DATE NULL,
  delivery_date DATE NULL,
  pickup_code INT NULL,
  FOREIGN KEY (status_id) REFERENCES OrderStatus(id),
  FOREIGN KEY (pickup_id) REFERENCES PickupPoint(id),
  FOREIGN KEY (client_id) REFERENCES User(id)
) ENGINE=InnoDB;
